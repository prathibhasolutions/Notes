\section{Fundamentals of Computers}

\subsection{Introduction to Computers}

\subsubsection{What is a Computer?}
A computer is an electronic device that accepts data as input, processes it according to a set of instructions (program), and produces information as output. The term "computer" is derived from the Latin word "computare" which means to calculate.

\textbf{Key Characteristics of Computers:}
\begin{itemize}[leftmargin=*]
    \item \textbf{Speed:} Computers can process millions of instructions per second
    \item \textbf{Accuracy:} Computers perform calculations with near-perfect accuracy
    \item \textbf{Storage:} Large amounts of data can be stored and retrieved quickly
    \item \textbf{Automation:} Can perform tasks automatically without human intervention
    \item \textbf{Versatility:} Can perform various types of tasks
    \item \textbf{Diligence:} Can work continuously without fatigue
\end{itemize}

\subsubsection{History of Computers}

\textbf{Generations of Computers:}

\begin{enumerate}
    \item \textbf{First Generation (1940-1956): Vacuum Tubes}
    \begin{itemize}
        \item Technology: Vacuum tubes for circuitry, magnetic drums for memory
        \item Examples: ENIAC, UNIVAC, EDVAC
        \item Characteristics: Very large in size, consumed lot of electricity, generated excessive heat, slow in speed, less reliable
    \end{itemize}
    
    \item \textbf{Second Generation (1956-1963): Transistors}
    \begin{itemize}
        \item Technology: Transistors replaced vacuum tubes
        \item Examples: IBM 1401, CDC 1604
        \item Characteristics: Smaller, faster, more reliable, less heat generation, less power consumption
        \item Programming: Assembly language, high-level languages like COBOL, FORTRAN
    \end{itemize}
    
    \item \textbf{Third Generation (1964-1971): Integrated Circuits}
    \begin{itemize}
        \item Technology: Integrated Circuits (ICs) - transistors on silicon chips
        \item Examples: IBM 360 series, Honeywell-6000 series
        \item Characteristics: Smaller size, faster, more reliable, affordable, commercial production started
        \item Development: Operating systems, multiprogramming
    \end{itemize}
    
    \item \textbf{Fourth Generation (1971-Present): Microprocessors}
    \begin{itemize}
        \item Technology: VLSI (Very Large Scale Integration) - thousands of ICs on single chip
        \item Examples: IBM PC, Apple Macintosh, modern desktop computers
        \item Characteristics: Portable, reliable, cheap, general-purpose
        \item Development: GUI, internet, multimedia
    \end{itemize}
    
    \item \textbf{Fifth Generation (Present-Future): Artificial Intelligence}
    \begin{itemize}
        \item Technology: ULSI (Ultra Large Scale Integration), AI, parallel processing
        \item Focus: Voice recognition, natural language processing, expert systems
        \item Examples: AI systems, quantum computers (in development)
    \end{itemize}
\end{enumerate}

\subsection{Computer Hardware and Software}

\subsubsection{Hardware}
Hardware refers to the physical components of a computer system that you can touch and see.

\textbf{Categories of Hardware:}
\begin{enumerate}
    \item \textbf{Input Devices:} Devices used to enter data
    \item \textbf{Output Devices:} Devices that display or output information
    \item \textbf{Processing Unit:} CPU (Central Processing Unit)
    \item \textbf{Storage Devices:} Devices that store data
\end{enumerate}

\subsubsection{Software}
Software is a set of programs, procedures, and documentation that performs specific tasks on a computer system.

\textbf{Types of Software:}

\begin{enumerate}
    \item \textbf{System Software:}
    \begin{itemize}
        \item Operating Systems (Windows, Linux, macOS)
        \item Device Drivers
        \item Utility Programs (Antivirus, Disk Cleanup)
        \item Language Translators (Compilers, Interpreters)
    \end{itemize}
    
    \item \textbf{Application Software:}
    \begin{itemize}
        \item Word Processors (MS Word)
        \item Spreadsheets (MS Excel)
        \item Database Management Systems (MS Access, Oracle)
        \item Graphics Software (Photoshop, CorelDraw)
        \item Web Browsers (Chrome, Firefox)
    \end{itemize}
\end{enumerate}

\subsection{Computer Memory}

\subsubsection{Memory Hierarchy}
Computer memory is organized in a hierarchy based on speed, cost, and capacity.

\textbf{Primary Memory (Main Memory):}

\begin{enumerate}
    \item \textbf{RAM (Random Access Memory)}
    \begin{itemize}
        \item Volatile memory - loses data when power is off
        \item Faster access speed
        \item Used for storing currently running programs and data
        \item Types: DRAM (Dynamic RAM), SRAM (Static RAM), DDR RAM
        \item Capacity: Typically 4GB to 32GB in modern computers
    \end{itemize}
    
    \item \textbf{ROM (Read Only Memory)}
    \begin{itemize}
        \item Non-volatile memory - retains data when power is off
        \item Stores permanent instructions (BIOS, firmware)
        \item Types: PROM (Programmable), EPROM (Erasable Programmable), EEPROM (Electrically Erasable Programmable)
    \end{itemize}
    
    \item \textbf{Cache Memory}
    \begin{itemize}
        \item Very fast memory between CPU and RAM
        \item Stores frequently accessed data
        \item Levels: L1 (fastest, smallest), L2, L3 (slowest, largest)
    \end{itemize}
\end{enumerate}

\begin{itemize}

    \item \textbf{Terabyte (TB):} 1024 GB

\end{itemize}

\subsection{Number Systems}

\subsubsection{Types of Number Systems}

\begin{enumerate}
    \item \textbf{Decimal Number System (Base 10)}
    \begin{itemize}
        \item Uses digits: 0-9
        \item Most commonly used by humans
        \item Example: 125, 456, 789
    \end{itemize}
    
    \item \textbf{Binary Number System (Base 2)}
    \begin{itemize}
        \item Uses digits: 0, 1
        \item Used by computers internally
        \item Example: 1010, 1101, 11110
    \end{itemize}
    
    \item \textbf{Octal Number System (Base 8)}
    \begin{itemize}
        \item Uses digits: 0-7
        \item Used in computer programming
        \item Example: 12, 45, 77
    \end{itemize}
    
    \item \textbf{Hexadecimal Number System (Base 16)}
    \begin{itemize}
        \item Uses digits: 0-9 and A-F (A=10, B=11, C=12, D=13, E=14, F=15)
        \item Used in memory addresses, color codes
        \item Example: 1A, FF, A5
    \end{itemize}
\end{enumerate}

\subsubsection{Number System Conversions}

\textbf{Decimal to Binary:}
\begin{itemize}
    \item Divide decimal number by 2
    \item Note the remainder
    \item Repeat until quotient becomes 0
    \item Write remainders in reverse order
    \item Example: (25)\textsubscript{10} = (11001)\textsubscript{2}
\end{itemize}

\textbf{Binary to Decimal:}
\begin{itemize}
    \item Multiply each bit by 2\textsuperscript{position}
    \item Sum all values
    \item Example: (1101)\textsubscript{2} = 1×2³ + 1×2² + 0×2¹ + 1×2⁰ = 8 + 4 + 0 + 1 = (13)\textsubscript{10}
\end{itemize}

\textbf{Decimal to Octal:}
\begin{itemize}
    \item Divide by 8, note remainders
    \item Example: (65)\textsubscript{10} = (101)\textsubscript{8}
\end{itemize}

\textbf{Decimal to Hexadecimal:}
\begin{itemize}
    \item Divide by 16, note remainders
    \item Example: (255)\textsubscript{10} = (FF)\textsubscript{16}
\end{itemize}

\subsection{Operating System Basics}

\subsubsection{What is an Operating System?}
An Operating System (OS) is system software that manages computer hardware and software resources and provides common services for computer programs. It acts as an interface between the user and computer hardware.

\textbf{Functions of Operating System:}
\begin{enumerate}
    \item \textbf{Process Management:} Managing running programs
    \item \textbf{Memory Management:} Allocating and deallocating memory
    \item \textbf{File Management:} Organizing files and directories
    \item \textbf{Device Management:} Managing input/output devices
    \item \textbf{Security:} Protecting system from unauthorized access
    \item \textbf{User Interface:} Providing interface for user interaction
\end{enumerate}

\subsubsection{Types of Operating Systems}

\begin{enumerate}
    \item \textbf{Single User, Single Tasking:} MS-DOS
    \item \textbf{Single User, Multi Tasking:} Windows, macOS
    \item \textbf{Multi User:} Unix, Linux
    \item \textbf{Real Time OS:} Used in embedded systems, industrial control
    \item \textbf{Mobile OS:} Android, iOS
    \item \textbf{Network OS:} Windows Server, Linux Server
\end{enumerate}

\subsubsection{Popular Operating Systems}

\textbf{Windows:}
\begin{itemize}
    \item Developed by Microsoft
    \item User-friendly GUI
    \item Versions: Windows 7, 8, 10, 11
    \item Widely used for desktop computers
\end{itemize}

\textbf{Linux:}
\begin{itemize}
    \item Open-source OS
    \item Free to use and modify
    \item Distributions: Ubuntu, Fedora, Debian, CentOS
    \item Popular in servers and programming
\end{itemize}

\textbf{macOS:}
\begin{itemize}
    \item Developed by Apple
    \item Used in Apple Mac computers
    \item Known for design and performance
\end{itemize}

\textbf{Mobile Operating Systems:}
\begin{itemize}
    \item \textbf{Android:} Open-source, Google, most popular mobile OS
    \item \textbf{iOS:} Apple, used in iPhone and iPad
\end{itemize}

\subsubsection{Basic OS Operations}

\textbf{File Management:}
\begin{itemize}
    \item Creating, copying, moving, deleting files and folders

    \item \textbf{Present:} IoT, cloud computing, 5G

\subsection{Internet Connectivity}

        \item Speed: 56 Kbps (very slow)
        \item Cannot use phone and internet simultaneously
        \item Obsolete technology
    \end{itemize}

    \begin{itemize}

    \item \textbf{Wireless Connection}
    \begin{itemize}
        \item \textbf{Wi-Fi:} Wireless local area network, range 30-50 meters
        \item \textbf{Mobile Data:} 3G, 4G, 5G cellular networks
        \item \textbf{Satellite:} Internet via satellite, for remote areas

    \item Fill in details:
    \begin{itemize}
        \item \textbf{To:} Recipient's email address
        \item \textbf{Cc (Carbon Copy):} Send copy to others (everyone sees)
        \item \textbf{Bcc (Blind Carbon Copy):} Send copy to others (hidden from other recipients)
        \item \textbf{Subject:} Brief description of email
        \item \textbf{Body:} Main message content
    \end{itemize}
    \item Format text (bold, italic, underline, font, color)
    \item Attach files if needed
    \item Click "Send"
\end{enumerate}

\textbf{Email Writing Tips:}
\begin{itemize}
    \item Clear and descriptive subject line
    \item Professional greeting (Dear Sir/Madam, Hi John)
    \item Keep message concise and clear
    \item Use proper grammar and spelling
    \item Professional tone for formal emails
    \item End with appropriate closing (Regards, Best wishes, Thank you)
    \item Include signature if needed
    \item Proofread before sending
\end{itemize}

\subsection{Internet Security and Safety}

\subsubsection{Online Threats}

\begin{enumerate}
    \item \textbf{Virus:} Malicious program that replicates and damages system
    \item \textbf{Malware:} Software designed to harm computer
    \item \textbf{Phishing:} Fake emails/websites to steal information
    \item \textbf{Spam:} Unwanted bulk emails
    \item \textbf{Ransomware:} Locks files and demands payment
    \item \textbf{Spyware:} Secretly monitors user activity
    \item \textbf{Identity Theft:} Stealing personal information
    \item \textbf{Hacking:} Unauthorized access to systems
\end{enumerate}

\subsubsection{Security Best Practices}

\textbf{Password Security:}
\begin{itemize}
    \item Use strong, unique passwords
    \item Don't share passwords
    \item Change passwords regularly
    \item Use password manager
    \item Enable two-factor authentication (2FA)
\end{itemize}

\textbf{Safe Browsing:}
\begin{itemize}
    \item Check website security (HTTPS, padlock icon)
    \item Don't click suspicious links
    \item Avoid downloading from untrusted sites
    \item Use secure Wi-Fi networks
    \item Clear browser cache and cookies regularly
\end{itemize}

\textbf{Email Security:}
\begin{itemize}
    \item Don't open emails from unknown senders
    \item Don't click links in suspicious emails
    \item Verify sender before sharing information
    \item Don't download unknown attachments
    \item Report spam and phishing
\end{itemize}

\textbf{Protection Tools:}
\begin{itemize}
    \item Install antivirus software (Windows Defender, Norton, McAfee)
    \item Keep software updated
    \item Use firewall
    \item Regular system scans
    \item Backup important data
\end{itemize}

\textbf{Personal Information:}
\begin{itemize}
    \item Don't share personal details publicly
    \item Be cautious on social media
    \item Read privacy policies
    \item Use privacy settings
    \item Monitor bank and credit card statements
\end{itemize}
